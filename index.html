<script>

function updateResult() {

    let offset = parseFloat(document.getElementById('offset').value);

    let distance = parseFloat(document.getElementById('distance').value);

    let clickMOA = parseFloat(document.getElementById('clickMOA').value);


    let mmPerMOA = 29.1 * (distance / 100);

    let clicks = Math.round(offset / (mmPerMOA * clickMOA));


    document.getElementById('result').innerHTML =

        `Result: ${clicks} clicks<br>1 MOA = ${mmPerMOA.toFixed(2)} mm @ ${distance} m`;

}


// Synchronizacja sliderów i pól input

const offsetInput = document.getElementById('offset');

const offsetSlider = document.getElementById('offsetSlider');

offsetInput.addEventListener('input', () => {

    offsetSlider.value = offsetInput.value;

    updateResult();

});

offsetSlider.addEventListener('input', () => {

    offsetInput.value = offsetSlider.value;

    updateResult();

});


const distanceInput = document.getElementById('distance');

const distanceSlider = document.getElementById('distanceSlider');

distanceInput.addEventListener('input', () => {

    distanceSlider.value = distanceInput.value;

    updateResult();

});

distanceSlider.addEventListener('input', () => {

    distanceInput.value = distanceSlider.value;

    updateResult();

});


document.getElementById('clickMOA').addEventListener('change', updateResult);


// Początkowe wyliczenie

updateResult();

</script>
